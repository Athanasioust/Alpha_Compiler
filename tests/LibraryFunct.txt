// Math functions
x = 16;
y = sqrt(x);
print("sqrt(16) = ", y);

angle = 0;
sin_val = sin(angle);
cos_val = cos(angle);
print("sin(0) = ", sin_val);
print("cos(0) = ", cos_val);

print("typeof([1,2,3]) = ", typeof(arr));
print("typeof(function) = ", typeof(func));
