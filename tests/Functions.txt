// Simple function
function add(x, y) {
    return x + y;
}

// Recursive function
function factorial(n) {
    if (n <= 1) {
        return 1;
    } else {
        return n * factorial(n - 1);
    }
}

// Function with local variables
function compute(a, b) {
    local sum = a + b;
    local diff = a - b;
    local prod = a * b;
    
    return sum + diff + prod;
}

// Function calls
result1 = add(5, 3);
result2 = factorial(5);
result3 = compute(10, 5);

print("add(5, 3) = ", result1);
print("factorial(5) = ", result2);
print("compute(10, 5) = ", result3);
