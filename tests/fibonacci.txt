// Iterative Fibonacci
function fib_iterative(n) {
    if (n <= 1) {
        return n;
    }
    
    local a = 0;
    local b = 1;
    local i;
    
    for (i = 2; i <= n; i++) {
        local temp = a + b;
        a = b;
        b = temp;
    }
    
    return b;
}

// Recursive Fibonacci
function fib_recursive(n) {
    if (n <= 1) {
        return n;
    }
    return fib_recursive(n - 1) + fib_recursive(n - 2);
}

// Test both versions
n = 10;
print("Fibonacci(", n, ") iterative = ", fib_iterative(n));
print("Fibonacci(", n, ") recursive = ", fib_recursive(n));

// Print Fibonacci sequence
print("Fibonacci sequence:");
for (i = 0; i < 10; i++) {
    print("fib(", i, ") = ", fib_iterative(i));
}
