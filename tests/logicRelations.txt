// Boolean values
a = true;
b = false;

// Logical operations
c = a and b;
d = a or b;
e = not a;

print("true and false = ", c);
print("true or false = ", d);
print("not true = ", e);

// Relational operations
x = 10;
y = 20;

if (x < y and y > 0) {
    print("x < y and y > 0");
}

if (x == 10 or y == 30) {
    print("x is 10 or y is 30");
}

// Complex logical expressions
result = (x > 5 and x < 15) or (y > 25 and y < 35);
print("Complex logical result = ", result);
