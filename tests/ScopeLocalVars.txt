// Global variable
global_var = 100;

function testScope() {
    // Local variable with same name
    local global_var = 200;
    local local_var = 300;
    
    print("Inside function: global_var = ", global_var);
    print("Inside function: local_var = ", local_var);
    
    // Accessing global explicitly
    print("Global variable: ", ::global_var);
}

testScope();
print("Outside function: global_var = ", global_var);
